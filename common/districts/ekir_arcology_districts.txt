district_arcology_science = 	{

	base_buildtime = 600
	is_capped_by_modifier = no

	show_on_uncolonized = {
		uses_district_set = city_world
		from = {
			OR = {
				AND = {
					is_regular_empire = yes
				}
				AND = {
					is_machine_empire = yes
				}
			}
		}
	}

	potential = {
		uses_district_set = city_world
		owner = {
			exists = owner
			OR = {
				AND = {
					is_regular_empire = yes
				}
				AND = {
					is_machine_empire = yes
				}
			}
		}
	}

	resources = {
		category = planet_districts
		cost = {
			minerals = 1000
			exotic_gases = 50
		}
		upkeep = {
			energy = 5
			exotic_gases = 1
		}
	}

	planet_modifier = {
		planet_housing_add = 10
	}

	### Normal
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_regular_empire = yes }
		}
		job_researcher_add = 6
	}

	### Hivemind
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_hive_empire = yes }
		}
		job_brain_drone_add = 6
	}

	### Machine
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_machine_empire = yes }
		}
		job_calculator_add = 6
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_regular_empire = yes }
		}
		text = job_researcher_effect_desc
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_hive_empire = yes }
		}
		text = job_brain_drone_effect_desc
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_machine_empire = yes }
		}
		text = job_calculator_effect_desc
	}
}

	base_buildtime = 600
	is_capped_by_modifier = no

	show_on_uncolonized = {
		uses_district_set = city_world
		from = {
			OR = {
				AND = {
					is_regular_empire = yes
					OR = {
						is_spiritualist = no
						has_make_spiritualist_perk = no
					}
				}
				AND = {
					is_machine_empire = yes
					has_tradition = tr_virtuality_4
				}
			}
		}
	}

	potential = {
		uses_district_set = city_world

			exists = owner

				}

					is_machine_empire = yes
				}

district_arcology_religious = {
	base_buildtime = 600
	is_capped_by_modifier = no

	show_on_uncolonized = {
		uses_district_set = city_world
		from = {
			OR = {
				is_spiritualist = yes
				has_make_spiritualist_perk = yes
			}
		}
	}

	potential = {
		uses_district_set = city_world
		owner = {
			exists = owner
			OR = {
				is_spiritualist = yes
				has_make_spiritualist_perk = yes
			}
		}
	}

	conversion_ratio = 1
	convert_to = {
		district_nexus
		district_hive
		district_arcology_administrative
	}

	resources = {
		category = planet_districts
		cost = {
			minerals = 1000
			rare_crystals = 50
		}
		upkeep = {
			energy = 5
			rare_crystals = 1
		}
	}

	planet_modifier = {
		planet_housing_add = 10
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_megacorp = no
				is_death_cult_empire = no
				NOT = { has_origin = origin_cybernetic_creed }
			}
		}
		modifier = {
			job_priest_add = 6
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_megacorp = yes
				is_death_cult_empire = no
				NOT = { has_origin = origin_cybernetic_creed }
			}
		}
		modifier = {
			# Megacorp YES, Death Cult NO.
			job_preacher_add = 6
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_origin = origin_cybernetic_creed # mutually exclusive with Death Cults
			}
		}
		modifier = {
			job_haruspex_add = 6
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_death_cult_empire = yes
			}
		}
		modifier = {
			# Death Cult YES.
			job_death_priest_add = 3
			job_mortal_initiate_add = 3
		}
	}

	#Jobs from virtuality for IMs
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_megacorp = no
				is_death_cult_empire = no
				NOT = { has_origin = origin_cybernetic_creed }
			}
		}
		modifier = {
			job_priest_add = 3
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_megacorp = yes
				is_death_cult_empire = no
				NOT = { has_origin = origin_cybernetic_creed }
			}
		}
		modifier = {
			# Megacorp YES, Death Cult NO.
			job_preacher_add = 3
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_death_cult_empire = yes
			}
		}
		modifier = {
			# Death Cult YES.
			job_death_priest_add = 2
			job_mortal_initiate_add = 1
		}
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = {
				is_megacorp = no
				is_death_cult_empire = no
				NOT = { has_origin = origin_cybernetic_creed }
			}
		}
		text = job_priest_effect_desc
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = {
				is_megacorp = yes
				is_death_cult_empire = no
				NOT = { has_origin = origin_cybernetic_creed }
			}
		}
		text = job_preacher_effect_desc
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = {
				has_origin = origin_cybernetic_creed # mutually exclusive with Death Cults
			}
		}
		text = job_haruspex_effect_desc
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = {
				is_death_cult_empire = yes # mutually exclusive with 'origin_cybernetic_creed'
			}
		}
		text = jobs_death_priest_and_mortal_initiate_effect_desc
	}
}
